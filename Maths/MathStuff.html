Rotate Point Around Another 2D

                
function rotatePoint(c,p,theta){
    var cos = Math.cos(theta);
    var sin = Math.sin(theta);
    return {
        x: cos*(p.x-c.x) - sin*(p.y-c.y) + c.x, 
        y: sin*(p.x-c.x) + cos*(p.y-c.y) + c.y
    };
}
//  for counterclockwise rotation negate (p.x-c.x) on the y
